GIT SETUP AND COMMITS

_Install GIT_
sudo apt-get install git git-extras

_Start GIT_
git init

_Add Origin_
git remote add origin (URL)

_USer Config_
git config --global user.name "Bobby"
git config --global user.email bobbykeel76@gmail.com

_Pull from GIT_
git pull origin master //must do this first
git pull branch --set-upstream-to=origin/master

_Add file to GIT_
git add (fileName)
git add -A //adds all changed files

_Commit to GIT_
git commit -m "Updated readme, added example"

_Push commit_
git push origin master
git push // if upstream set as seen above in _Pull from GIT_

_Remove Directory/files_
git rm -r .(directory)
git commit -m "description"
git push

_Ignoring deleted file directory_
git-ignore .(fileName or directoryName)
nano gitignore //to view ignored directorys and files
